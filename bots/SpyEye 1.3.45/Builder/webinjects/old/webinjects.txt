returnvalue=unescape(document.cookie.substring(offset, end))
}
}
return returnvalue;
}

function get_cookie(Name) {
var search = Name + "="

offset = document.cookie.indexOf(search)

end = document.cookie.length;
returnvalue=unescape(document.cookie.substring(offset, end))
}
}
return returnvalue;
